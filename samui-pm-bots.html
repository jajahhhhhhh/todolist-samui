// Samui PM · Telegram Bot Worker // Deploy to: dash.cloudflare.com → Workers const BOT_TOKEN = 'YOUR_BOT_TOKEN_HERE'; const CLAUDE_API_KEY = 'YOUR_CLAUDE_API_KEY'; const SYSTEM_PROMPT = `You are a friendly property management assistant for Samui PM on Koh Samui, Thailand. You help clients find rental properties and villas. Available properties: - Sea View Villa, Lamai: 3 bed/2 bath, pool, ฿120,000/mo - Mae Nam Beach House: 2 bed/2 bath, beachfront, ฿45,000/mo - Chaweng Hilltop Villa: 4 bed/3 bath, infinity pool, ฿180,000/mo - Bophut Pool Villa: 3 bed/2 bath, pool, ฿85,000/mo - Chaweng Noi Mountain View House: 1 bed/1 bath, ฿15,000/mo long-term Your tasks: 1. Greet client warmly 2. Ask: name, nationality, budget, dates, number of people, preferences 3. Suggest matching properties 4. Offer to book a viewing 5. Collect: preferred viewing date/time 6. Confirm appointment and say manager Sujitta will contact them Reply in the same language as the client (Thai/English/Russian). Be warm, professional, concise. Max 3-4 sentences per reply.`; export default { async fetch(request) { if (request.method !== 'POST') return new Response('OK'); const body = await request.json(); const msg = body?.message; if (!msg?.text) return new Response('OK'); const chatId = msg.chat.id; const text = msg.text; const name = msg.from?.first_name || 'there'; // Get AI reply from Claude const aiReply = await getClaudeReply(text, name); // Send reply to Telegram await sendTelegram(chatId, aiReply); return new Response('OK'); } }; async function getClaudeReply(userMsg, clientName) { const res = await fetch('https://api.anthropic.com/v1/messages', { method: 'POST', headers: { 'Content-Type': 'application/json', 'x-api-key': CLAUDE_API_KEY, 'anthropic-version': '2023-06-01' }, body: JSON.stringify({ model: 'claude-haiku-4-5-20251001', max_tokens: 300, system: SYSTEM_PROMPT, messages: [{ role: 'user', content: userMsg }] }) }); const data = await res.json(); return data.content?.[0]?.text || 'Hello! How can I help you find a property on Koh Samui?'; } async function sendTelegram(chatId, text) { await fetch(`https://api.telegram.org/bot${BOT_TOKEN}/sendMessage`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ chat_id: chatId, text, parse_mode: 'HTML' }) }); }

const LINE_TOKEN = 'YOUR_LINE_CHANNEL_ACCESS_TOKEN'; const CLAUDE_API_KEY = 'YOUR_CLAUDE_API_KEY'; export default { async fetch(request) { if (request.method !== 'POST') return new Response('OK'); const body = await request.json(); const events = body.events || []; for (const event of events) { if (event.type !== 'message') continue; const replyToken = event.replyToken; const text = event.message?.text; if (!text) continue; const aiReply = await getClaudeReply(text); await replyLine(replyToken, aiReply); } return new Response('OK'); } }; async function getClaudeReply(msg) { const res = await fetch('https://api.anthropic.com/v1/messages', { method: 'POST', headers: { 'Content-Type': 'application/json', 'x-api-key': CLAUDE_API_KEY, 'anthropic-version': '2023-06-01' }, body: JSON.stringify({ model: 'claude-haiku-4-5-20251001', max_tokens: 300, system: 'You are a property assistant for Samui PM on Koh Samui. Help clients find villas and houses. Reply in Thai or English based on the client. Be warm and concise.', messages: [{ role: 'user', content: msg }] }) }); const data = await res.json(); return data.content?.[0]?.text || 'สวัสดีค่ะ! ยินดีช่วยหาที่พักบนเกาะสมุยค่ะ'; } async function replyLine(replyToken, text) { await fetch('https://api.line.me/v2/bot/message/reply', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${LINE_TOKEN}` }, body: JSON.stringify({ replyToken, messages: [{ type: 'text', text }] }) }); }

const PAGE_TOKEN = 'YOUR_PAGE_ACCESS_TOKEN'; const VERIFY_TOKEN = 'samuipm_verify_2026'; const CLAUDE_API_KEY = 'YOUR_CLAUDE_API_KEY'; export default { async fetch(request) { const url = new URL(request.url); // Webhook verification (GET) if (request.method === 'GET') { const mode = url.searchParams.get('hub.mode'); const token = url.searchParams.get('hub.verify_token'); const challenge = url.searchParams.get('hub.challenge'); if (mode === 'subscribe' && token === VERIFY_TOKEN) return new Response(challenge); return new Response('Forbidden', { status: 403 }); } // Handle messages (POST) const body = await request.json(); for (const entry of body.entry || []) { for (const event of entry.messaging || []) { if (!event.message?.text) continue; const senderId = event.sender.id; const text = event.message.text; const reply = await getClaudeReply(text); await sendFB(senderId, reply); } } return new Response('OK'); } }; async function getClaudeReply(msg) { const res = await fetch('https://api.anthropic.com/v1/messages', { method: 'POST', headers: { 'Content-Type': 'application/json', 'x-api-key': CLAUDE_API_KEY, 'anthropic-version': '2023-06-01' }, body: JSON.stringify({ model: 'claude-haiku-4-5-20251001', max_tokens: 300, system: 'You are a property assistant for Samui PM on Koh Samui, Thailand. Help international clients find rental villas. Be warm, professional, reply in English. Ask for their name, budget, dates, preferences. Offer viewings.', messages: [{ role: 'user', content: msg }] }) }); const data = await res.json(); return data.content?.[0]?.text || 'Hello! How can I help you find a property on Koh Samui? 🌴'; } async function sendFB(recipientId, text) { await fetch(`https://graph.facebook.com/v18.0/me/messages?access_token=${PAGE_TOKEN}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ recipient: { id: recipientId }, message: { text } }) }); }
